155 static BSP430_CORE_INLINE int
156 iBSP430sensorsMCP9808tempFromRaw_cK (uint16_t raw)
157 {
158  long v = 10 * (raw & BSP430_SENSORS_MCP9808_TEMP_MASK);
159  v += BSP430_SENSORS_MCP9808_OFFSET_Kd160;
160  /* Convert to K/1600 then divide by 16 to get K/100 = cK. */
161  v = (v * 10) / 16;
162  if (BSP430_SENSORS_MCP9808_TEMP_SIGN & raw) {
163  v = -v;
164  }
165  return (int)v;
166 }
